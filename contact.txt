        // Firebase logic for Contact Us form
        document.getElementById('contactForm').addEventListener('submit', function(e) {
            e.preventDefault();

            const name = document.getElementById('name').value;
            const email = document.getElementById('email').value;
            const message = document.getElementById('message').value;

            // Send contact form data to Firebase
            fetch('https://weight-scale-w13-be5a0-default-rtdb.firebaseio.com/contactUs.json?auth=AIzaSyCDw1Wj7XKAW3ulebJgoLrIXbMM48UeKEY', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    name: name,
                    email: email,
                    message: message
                })
            })
            .then(response => response.json())
            .then(data => {
                alert('Your message has been sent successfully!');
                document.getElementById('contactForm').reset();
            })
            .catch(error => {
                console.error('Error:', error);
            });
        });

        // Firebase logic for Demo Request form
        document.getElementById('demoForm').addEventListener('submit', function(e) {
            e.preventDefault();

            const company = document.getElementById('company').value;
            const demoEmail = document.getElementById('demo-email').value;
            const demoDetails = document.getElementById('demo-details').value;

            // Send demo request data to Firebase
            fetch('https://weight-scale-w13-be5a0-default-rtdb.firebaseio.com/demoRequests.json?auth=AIzaSyCDw1Wj7XKAW3ulebJgoLrIXbMM48UeKEY', {
                method: 'POST',

                    'Content-Type': 'application/json'
                },

<script>
    // OEE Calculation
    function calculateOEE() {
        let availability = parseFloat(document.getElementById("availability").value) / 100;
        let performance = parseFloat(document.getElementById("performance").value) / 100;
        let quality = parseFloat(document.getElementById("quality").value) / 100;
        let oee = (availability * performance * quality) * 100;
        document.getElementById("oeeResult").innerText = oee.toFixed(2);
    }

    // ROI Calculation
    function calculateROI() {
        let gain = parseFloat(document.getElementById("gain").value);
        let cost = parseFloat(document.getElementById("cost").value);
        let roi = ((gain - cost) / cost) * 100;
        document.getElementById("roiResult").innerText = roi.toFixed(2);
    }

    // Cycle Time Calculation
    function calculateCycleTime() {
        let totalTime = parseFloat(document.getElementById("totalTime").value);
        let totalUnits = parseInt(document.getElementById("totalUnits").value);
        let cycleTime = totalTime / totalUnits;
        document.getElementById("cycleTimeResult").innerText = cycleTime.toFixed(2);
    }
</script>
